// and gate 

module top_module( 
    input a, 
    input b, 
    output out );
    assign out=a&b;
endmodule


/// nor gate

module top_module( 
    input a, 
    input b, 
    output out );
    assign out=~(a|b);
endmodule


////Xnor gate

module top_module( 
    input a, 
    input b, 
    output out );
    assign out=~(a^b);
endmodule
