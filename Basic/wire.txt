module top_module( 
    input a,b,c,
    output w,x,y,z );
assign w = a;
assign x = b;
assign y = b;
assign z = c;

endmodule



/// wire2

`default_nettype none
module top_module(
    input a,
    input b,
    input c,
    input d,
    output out,
    output out_n
); 
wire x,y;
    assign x = a & b;        
    assign y = c & d;        
    assign out   = x | y;    
    assign out_n = ~out;   
endmodule
