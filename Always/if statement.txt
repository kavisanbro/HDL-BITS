module top_module (
    input a, 
    input b, 
    input sel_b1, 
    input sel_b2,
    output wire out_assign,
    output reg out_always
);

    // Method 1: assign (continuous assignment with conditional operator)
    assign out_assign = (sel_b1 & sel_b2) ? b : a;

    // Method 2: always block (procedural if statement)
    always @(*) begin
        if (sel_b1 & sel_b2)
            out_always = b;
        else
            out_always = a;
    end

endmodule
